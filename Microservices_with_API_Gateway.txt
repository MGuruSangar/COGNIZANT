  port: 8080

  application:
    name: api-gateway

  cloud:
    gateway:
      routes:
        - id: account-service
          uri: http://localhost:8081
          predicates:
            - Path=/account/**
        - id: loan-service
          uri: http://localhost:8082
          predicates:
            - Path=/loan/**
